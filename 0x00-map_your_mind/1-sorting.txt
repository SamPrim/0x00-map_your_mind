function SORTIN_NUMBER()

start

  Read a, b, c;

  if a > b and a < c
    write a
    if b > c
      write b
    if c > b
      write c
  else if b > a and b >c
    write b
    if a > c
      write a
    if c > a
      write c
  if c > a and c > b
    write c
    if a > b
      write a
    if b > a
      write b

end
